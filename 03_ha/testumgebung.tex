                                                      & = \lim_{k\to\infty}\left(\frac{(k+1)-k}{\sqrt{k+1}+\sqrt{k}}\right) \\
                                                      & = \lim_{k\to\infty}\left(\frac{1}{\sqrt{k+1}+\sqrt{k}}\right) \\
                                                      & = 0
\end{align*}
